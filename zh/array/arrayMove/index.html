import { AnyArrayMap } from 'types';

export function arrayMove(array: AnyArrayMap, from: number, to: number): AnyArrayMap {
  array = array.slice();
  array.splice(to < 0 ? array.length + to : to, 0, array.splice(from, 1)[0]);
  return array;
}
